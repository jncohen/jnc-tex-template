% =======================================================
% 3b. JOURNAL-STYLE FLOAT DEFERRAL
%
% Activated by: journal_style: true  (YAML)
%
% All figure and table float environments are deferred to after
% the bibliography. A bold, centred placeholder marker is printed
% in the text body wherever each float originally appeared.
%
% Uses the endfloat package (ships with TeXLive / MiKTeX).
% Requires TWO compilation passes (the same two-pass requirement
% as BibTeX / biber).
%
% Output order:
%   1. Main text (with [Insert ...] markers)
%   2. Bibliography
%   3. Tables (one per page)
%   4. Figures (one per page)
%
% IMPORTANT — longtable (Pandoc's default environment for plain
% markdown table syntax) is a non-float and is NOT captured by
% endfloat. For full deferral in journal_style mode, generate
% tables as table floats rather than longtables:
%   knitr / R:   knitr::kable(..., format="latex", longtable=FALSE)
%   Raw LaTeX:   \begin{table}[p] ... \end{table}
% See README § "Journal-Style Submission Mode" for details.
%
% Known incompatibilities:
%   float [H] specifier — ignored; endfloat defers all floats.
%   linenumbers + journal_style — may cause layout artefacts;
%     avoid combining the two options.
% =======================================================
$if(journal_style)$
\usepackage[nofiglist,notablist,tablesfirst]{endfloat}

%% One float per page (standard journal convention)
\renewcommand{\efloatseparator}{\clearpage}

%% Bold centred placeholder for deferred figures
\renewcommand{\figureplace}{%
  \vspace{\baselineskip}%
  \begin{center}%
    \textbf{[Insert Figure \thefigure\ about here]}%
  \end{center}%
  \vspace{\baselineskip}%
}

%% Bold centred placeholder for deferred tables
\renewcommand{\tableplace}{%
  \vspace{\baselineskip}%
  \begin{center}%
    \textbf{[Insert Table \thetable\ about here]}%
  \end{center}%
  \vspace{\baselineskip}%
}
$endif$

%% Shared author-block macro.
%%
%% Author names come from the for(author) loop — Quarto passes author.name
%% through its normalize-author.lua filter. All other author sub-fields
%% (department, institution, email, orcid, etc.) are stripped by that filter
%% and never reach the Pandoc template.
%%
%% Solution: supply contact details via top-level author_* YAML variables,
%% which Quarto does not process and therefore passes through unchanged.
%%
%%   author_dept:   Department of Sociology
%%   author_inst:   Queens College, City University of New York
%%   author_addr:   65-30 Kissena Blvd., Queens, NY 11367, USA
%%   author_email:  joseph.cohen@qc.cuny.edu
%%   author_web:    https://jncohen.commons.gc.cuny.edu
%%   author_orcid:  0000-0002-6197-4453
%%
\newcommand{\jncauthorblock}{%
$if(author)$
\ifjncanon\else
$for(author)$
$if(author.name)$\textbf{$author.name$}$else$\textbf{$author$}$endif$\par
$endfor$
{\small\setlength{\parskip}{1pt}%
$if(author_dept)$$author_dept$\par$endif$%
$if(author_inst)$$author_inst$\par$endif$%
$if(author_addr)$$author_addr$\par$endif$%
$if(author_email)$\texttt{$author_email$}\par$endif$%
$if(author_web)$$author_web$\par$endif$%
$if(author_orcid)$ORCID: $author_orcid$\par$endif$%
}%
\fi
$endif$
}
